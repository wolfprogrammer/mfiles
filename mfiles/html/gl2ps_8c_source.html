01207         - plane[0] * prim->verts[0].xyz[0] 
01208         - plane[1] * prim->verts[0].xyz[1] 
01209         - plane[2] * prim->verts[0].xyz[2];
01210     }
01211     break;
01212   case GL2PS_POINT :
01213   case GL2PS_PIXMAP :
01214   case GL2PS_TEXT :
01215   case GL2PS_SPECIAL :
01216   case GL2PS_IMAGEMAP:
01217     plane[0] = plane[1] = 0.0F;
01218     plane[2] = 1.0F;
01219     plane[3] = -prim->verts[0].xyz[2];
01220     break;
01221   default :
01222     gl2psMsg(GL2PS_ERROR, "Unknown primitive type in BSP tree");
01223     plane[0] = plane[1] = plane[3] = 0.0F;
01224     plane[2] = 1.0F;
01225     break;
01226   }
01227 }

01229 static void gl2psCutEdge(GL2PSvertex *a, GL2PSvertex *b, GL2PSplane plane,
01230                          GL2PSvertex *c)
01231 {
01232   GL2PSxyz v;
01233   GLfloat sect;
01234 
01235   v[0] = b->xyz[0] - a->xyz[0];
01236   v[1] = b->xyz[1] - a->xyz[1];
01237   v[2] = b->xyz[2] - a->xyz[2];
01238 
01239   sect = - gl2psComparePointPlane(a->xyz, plane) / gl2psPsca(plane, v);
01240 
01241   c->xyz[0] = a->xyz[0] + v[0] * sect;
01242   c->xyz[1] = a->xyz[1] + v[1] * sect;
01243   c->xyz[2] = a->xyz[2] + v[2] * sect;
01244   
01245   c->rgba[0] = (1 - sect) * a->rgba[0] + sect * b->rgba[0];
01246   c->rgba[1] = (1 - sect) * a->rgba[1] + sect * b->rgba[1];
01247   c->rgba[2] = (1 - sect) * a->rgba[2] + sect * b->rgba[2];
01248   c->rgba[3] = (1 - sect) * a->rgba[3] + sect * b->rgba[3];
01249 }

01360     if(d[0] > GL2PS_EPSILON)       type = GL2PS_IN_BACK_OF;
01361     else if(d[0] < -GL2PS_EPSILON) type = GL2PS_IN_FRONT_OF;

01364   default :

01380         if(d[i] > GL2PS_EPSILON){

01382           gl2psAddIndex(out0, out1, &out, i, j);
01383           type = GL2PS_SPANNING;
01384         }
01385         gl2psAddIndex(in0, in1, &in, j, -1);
01386       }
01387       else{
01388         gl2psAddIndex(in0, in1, &in, j, -1);
01389         gl2psAddIndex(out0, out1, &out, j, -1);
01390       }
01391     }
01392     break;
01393   }
01394 
01395   if(type == GL2PS_SPANNING){

01399     gl2psCreateSplitPrimitive(prim, plane, *front, in, in0, in1);
01400   }
01401 
01402   return type;
01403 }

01685 static void gl2psRescaleAndOffset()
01686 {
01687   GL2PSprimitive *prim;
01688   GLfloat minZ, maxZ, rangeZ, scaleZ;
01689   GLfloat factor, units, area, dZ, dZdX, dZdY, maxdZ;
01690   int i, j;
01691 
01692   if(!gl2psListNbr(gl2ps->primitives))
01693     return;
01694 
01695   /* get z-buffer range */
01696   prim = *(GL2PSprimitive**)gl2psListPointer(gl2ps->primitives, 0);
01697   minZ = maxZ = prim->verts[0].xyz[2];
01698   for(i = 1; i < prim->numverts; i++){
01699     if(prim->verts[i].xyz[2] < minZ) minZ = prim->verts[i].xyz[2];
01700     if(prim->verts[i].xyz[2] > maxZ) maxZ = prim->verts[i].xyz[2];
01701   }
01702   for(i = 1; i < gl2psListNbr(gl2ps->primitives); i++){
01703     prim = *(GL2PSprimitive**)gl2psListPointer(gl2ps->primitives, i);
01704     for(j = 0; j < prim->numverts; j++){
01705       if(prim->verts[j].xyz[2] < minZ) minZ = prim->verts[j].xyz[2];
01706       if(prim->verts[j].xyz[2] > maxZ) maxZ = prim->verts[j].xyz[2];
01707     }
01708   }
01709   rangeZ = (maxZ - minZ);
01710 
01711   /* rescale z-buffer coordinate in [0,GL2PS_ZSCALE], to make it of
01712      the same order of magnitude as the x and y coordinates */
01713   scaleZ = GL2PS_ZERO(rangeZ) ? GL2PS_ZSCALE : (GL2PS_ZSCALE / rangeZ);
01714   /* avoid precision loss (we use floats!) */
01715   if(scaleZ > 100000.F) scaleZ = 100000.F;
01716 
01717   /* apply offsets */
01718   for(i = 0; i < gl2psListNbr(gl2ps->primitives); i++){
01719     prim = *(GL2PSprimitive**)gl2psListPointer(gl2ps->primitives, i);
01720     for(j = 0; j < prim->numverts; j++){
01721       prim->verts[j].xyz[2] = (prim->verts[j].xyz[2] - minZ) * scaleZ;
01722     }
01723     if((gl2ps->options & GL2PS_SIMPLE_LINE_OFFSET) &&
01724        (prim->type == GL2PS_LINE)){
01725       if(gl2ps->sort == GL2PS_SIMPLE_SORT){
01726         prim->verts[0].xyz[2] -= GL2PS_ZOFFSET_LARGE;
01727         prim->verts[1].xyz[2] -= GL2PS_ZOFFSET_LARGE;
01728       }
01729       else{
01730         prim->verts[0].xyz[2] -= GL2PS_ZOFFSET;
01731         prim->verts[1].xyz[2] -= GL2PS_ZOFFSET;
01732       }
01733     }
01734     else if(prim->offset && (prim->type == GL2PS_TRIANGLE)){
01735       factor = gl2ps->offset[0];
01736       units = gl2ps->offset[1];
01737       area = 
01738         (prim->verts[1].xyz[0] - prim->verts[0].xyz[0]) * 
01739         (prim->verts[2].xyz[1] - prim->verts[1].xyz[1]) - 
01740         (prim->verts[2].xyz[0] - prim->verts[1].xyz[0]) * 
01741         (prim->verts[1].xyz[1] - prim->verts[0].xyz[1]);
01742       dZdX = 
01743         ((prim->verts[2].xyz[1] - prim->verts[1].xyz[1]) *
01744          (prim->verts[1].xyz[2] - prim->verts[0].xyz[2]) -
01745          (prim->verts[1].xyz[1] - prim->verts[0].xyz[1]) *
01746          (prim->verts[2].xyz[2] - prim->verts[1].xyz[2])) / area;
01747       dZdY = 
01748         ((prim->verts[1].xyz[0] - prim->verts[0].xyz[0]) *
01749          (prim->verts[2].xyz[2] - prim->verts[1].xyz[2]) -
01750          (prim->verts[2].xyz[0] - prim->verts[1].xyz[0]) *
01751          (prim->verts[1].xyz[2] - prim->verts[0].xyz[2])) / area;
01752       maxdZ = (GLfloat)sqrt(dZdX * dZdX + dZdY * dZdY);
01753       dZ = factor * maxdZ + units;
01754       prim->verts[0].xyz[2] += dZ;
01755       prim->verts[1].xyz[2] += dZ;
01756       prim->verts[2].xyz[2] += dZ;
01757     }
01758   }

01767 static GLint gl2psGetPlaneFromPoints(GL2PSxyz a, GL2PSxyz b, GL2PSplane plane)
01768 {

01770 
01771   plane[0] = b[1] - a[1];
01772   plane[1] = a[0] - b[0];

01774   plane[2] = 0.0F;
01775   if(!GL2PS_ZERO(n)){
01776     plane[0] /= n;
01777     plane[1] /= n;
01778     plane[3] = -plane[0]*a[0]-plane[1]*a[1]; 
01779     return 1;
01780   }
01781   else{
01782     plane[0] = -1.0F;
01783     plane[1] = 0.0F;
01784     plane[3] = a[0];
01785     return 0;
01786   }

01958 static void gl2psSplitPrimitive2D(GL2PSprimitive *prim,
01959                                   GL2PSplane plane, 
01960                                   GL2PSprimitive **front, 
01961                                   GL2PSprimitive **back)
01962 {
01963   /* cur will hold the position of the current vertex
01964      prev will hold the position of the previous vertex
01965      prev0 will hold the position of the vertex number 0
01966      v1 and v2 represent the current and previous vertices, respectively
01967      flag is set if the current vertex should be checked against the plane */
01968   GLint cur = -1, prev = -1, i, v1 = 0, v2 = 0, flag = 1, prev0 = -1;
01969   
01970   /* list of vertices that will go in front and back primitive */

01979       if(prim->numverts < 3) break;
01980       v1 = 0;
01981       v2 = prim->numverts-1;
01982       cur = prev0;
01983     }
01984     else if(flag){
01985       cur = gl2psCheckPoint(prim->verts[v1].xyz, plane);
01986       if(i == 0){
01987         prev0 = cur;
01988       }
01989     } 
01990     if(((prev == -1) || (prev == cur) || (prev == 0) || (cur == 0)) &&
01991        (i < prim->numverts)){
01992       if(cur == GL2PS_POINT_INFRONT){
01993         front_count++;
01994         front_list = (GL2PSvertex*)gl2psRealloc(front_list,
01995                                                 sizeof(GL2PSvertex)*front_count);
01996         front_list[front_count-1] = prim->verts[v1];
01997       }
01998       else if(cur == GL2PS_POINT_BACK){
01999         back_count++;
02000         back_list = (GL2PSvertex*)gl2psRealloc(back_list,
02001                                                sizeof(GL2PSvertex)*back_count);
02002         back_list[back_count-1] = prim->verts[v1];
02003       }
02004       else{
02005         front_count++;
02006         front_list = (GL2PSvertex*)gl2psRealloc(front_list,
02007                                                 sizeof(GL2PSvertex)*front_count);
02008         front_list[front_count-1] = prim->verts[v1];
02009         back_count++;
02010         back_list = (GL2PSvertex*)gl2psRealloc(back_list,
02011                                                sizeof(GL2PSvertex)*back_count);
02012         back_list[back_count-1] = prim->verts[v1];
02013       }
02014       flag = 1;
02015     }
02016     else if((prev != cur) && (cur != 0) && (prev != 0)){
02017       if(v1 != 0){
02018         v2 = v1-1;
02019         i--;
02020       }
02021       front_count++;
02022       front_list = (GL2PSvertex*)gl2psRealloc(front_list,
02023                                               sizeof(GL2PSvertex)*front_count);

02025                    &prim->verts[v1],
02026                    plane,
02027                    &front_list[front_count-1]);

02030                                              sizeof(GL2PSvertex)*back_count);

02032       flag = 0;
02033     }
02034     prev = cur;
02035   }
02036   *front = gl2psCreateSplitPrimitive2D(prim, front_count, front_list);
02037   *back = gl2psCreateSplitPrimitive2D(prim, back_count, back_list);
02038   gl2psFree(front_list);
02039   gl2psFree(back_list);
02040 }

02310         if(v == 2){
02311           if(GL_TRUE == boundary){
02312             if(!count && vtot == 2) flag = 1|2|4;
02313             else if(!count) flag = 2|4;
02314             else if(vtot == 2) flag = 1|2;
02315             else flag = 2;
02316           }
02317           else
02318             flag = 0;
02319           gl2psAddPolyPrimitive(GL2PS_TRIANGLE, 3, vertices, offset,
02320                                 pattern, factor, 1, flag);
02321           vertices[1] = vertices[2];
02322         }
02323         else
02324           v ++;
02325       }
02326       break;      
02327     case GL_BITMAP_TOKEN :
02328     case GL_DRAW_PIXEL_TOKEN :
02329     case GL_COPY_PIXEL_TOKEN :
02330       current ++;
02331       used --;
02332       i = gl2psGetVertex(&vertices[0], current);
02333       current += i;
02334       used    -= i;
02335       break;      
02336     case GL_PASS_THROUGH_TOKEN :
02337       switch((GLint)current[1]){
02338       case GL2PS_BEGIN_OFFSET_TOKEN : offset = 1; break;
02339       case GL2PS_END_OFFSET_TOKEN : offset = 0; break;
02340       case GL2PS_BEGIN_BOUNDARY_TOKEN : boundary = GL_TRUE; break;
02341       case GL2PS_END_BOUNDARY_TOKEN : boundary = GL_FALSE; break;
02342       case GL2PS_END_STIPPLE_TOKEN : pattern = factor = 0; break;
02343       case GL2PS_BEGIN_BLEND_TOKEN : gl2ps->blending = GL_TRUE; break;
02344       case GL2PS_END_BLEND_TOKEN : gl2ps->blending = GL_FALSE; break;
02345       case GL2PS_BEGIN_STIPPLE_TOKEN : 
02346         current += 2;
02347         used -= 2; 
02348         pattern = (GLushort)current[1]; 
02349         current += 2;
02350         used -= 2; 
02351         factor = (GLint)current[1]; 
02352         break;
02353       case GL2PS_SRC_BLEND_TOKEN : 
02354         current += 2; 
02355         used -= 2; 
02356         gl2ps->blendfunc[0] = (GLint)current[1];
02357         break;
02358       case GL2PS_DST_BLEND_TOKEN : 
02359         current += 2; 
02360         used -= 2; 
02361         gl2ps->blendfunc[1] = (GLint)current[1];
02362         break;
02363       case GL2PS_POINT_SIZE_TOKEN : 
02364         current += 2; 
02365         used -= 2; 
02366         psize = current[1];
02367         break;
02368       case GL2PS_LINE_WIDTH_TOKEN : 
02369         current += 2; 
02370         used -= 2; 
02371         lwidth = current[1];
02372         break;
02373       case GL2PS_IMAGEMAP_TOKEN :
02374         prim = (GL2PSprimitive *)gl2psMalloc(sizeof(GL2PSprimitive));
02375         prim->type = GL2PS_IMAGEMAP;
02376         prim->boundary = 0;
02377         prim->numverts = 4;
02378         prim->verts = (GL2PSvertex *)gl2psMalloc(4 * sizeof(GL2PSvertex));
02379         prim->culled = 0;
02380         prim->offset = 0;
02381         prim->pattern = 0;
02382         prim->factor = 0;
02383         prim->width = 1;
02384         
02385         node = (GL2PSimagemap*)gl2psMalloc(sizeof(GL2PSimagemap));
02386         node->image = (GL2PSimage*)gl2psMalloc(sizeof(GL2PSimage));
02387         node->image->type = 0;
02388         node->image->format = 0;
02389         node->next = NULL;
02390         
02391         if(gl2ps->imagemap_head == NULL)
02392           gl2ps->imagemap_head = node;
02393         else
02394           gl2ps->imagemap_tail->next = node;
02395         gl2ps->imagemap_tail = node;
02396         prim->data.image = node->image;
02397         
02398         current += 2; used -= 2;
02399         i = gl2psGetVertex(&prim->verts[0], &current[1]);
02400         current += i; used -= i;
02401         
02402         node->image->width = (GLint)current[2];
02403         current += 2; used -= 2;
02404         node->image->height = (GLint)current[2];
02405         prim->verts[0].xyz[0] = prim->verts[0].xyz[0] - (int)(node->image->width / 2) + 0.5;
02406         prim->verts[0].xyz[1] = prim->verts[0].xyz[1] - (int)(node->image->height / 2) + 0.5;
02407         for(i = 1; i < 4; i++){
02408           for(v = 0; v < 3; v++){
02409             prim->verts[i].xyz[v] = prim->verts[0].xyz[v];
02410             prim->verts[i].rgba[v] = prim->verts[0].rgba[v];
02411           }
02412           prim->verts[i].rgba[v] = prim->verts[0].rgba[v];
02413         }
02414         prim->verts[1].xyz[0] = prim->verts[1].xyz[0] + node->image->width;
02415         prim->verts[2].xyz[0] = prim->verts[1].xyz[0];
02416         prim->verts[2].xyz[1] = prim->verts[2].xyz[1] + node->image->height;
02417         prim->verts[3].xyz[1] = prim->verts[2].xyz[1];
02418 
02419         sizeoffloat = sizeof(GLfloat);
02420         v = 2 * sizeoffloat;
02421         vtot = node->image->height + node->image->height * 
02422           ((node->image->width-1)/8);
02423         node->image->pixels = (GLfloat*)gl2psMalloc(v + vtot);
02424         node->image->pixels[0] = prim->verts[0].xyz[0];
02425         node->image->pixels[1] = prim->verts[0].xyz[1];
02426         
02427         for(i = 0; i < vtot; i += sizeoffloat){
02428           current += 2; used -= 2;
02429           if((vtot - i) >= 4)
02430             memcpy(&(((char*)(node->image->pixels))[i + v]), &(current[2]), sizeoffloat);
02431           else
02432             memcpy(&(((char*)(node->image->pixels))[i + v]), &(current[2]), vtot - i);
02433         }
02434         current++; used--;
02435         gl2psListAdd(gl2ps->primitives, &prim);
02436         break;
02437       case GL2PS_DRAW_PIXELS_TOKEN :
02438       case GL2PS_TEXT_TOKEN :
02439         if(auxindex < gl2psListNbr(gl2ps->auxprimitives))
02440           gl2psListAdd(gl2ps->primitives, 
02441                        gl2psListPointer(gl2ps->auxprimitives, auxindex++));
02442         else
02443           gl2psMsg(GL2PS_ERROR, "Wrong number of auxiliary tokens in buffer");
02444         break;
02445       }
02446       current += 2; 
02447       used -= 2; 
02448       break;      
02449     default :
02450       gl2psMsg(GL2PS_WARNING, "Unknown token in buffer");
02451       current ++;
02452       used --;
02453       break;
02454     }
02455   }
02456 
02457   gl2psListReset(gl2ps->auxprimitives);
02458 }

03005 static void gl2psParseStipplePattern(GLushort pattern, GLint factor, 
03006                                      int *nb, int array[10])
03007 {
03008   int i, n;
03009   int on[8] = {0, 0, 0, 0, 0, 0, 0, 0};
03010   int off[8] = {0, 0, 0, 0, 0, 0, 0, 0};
03011   char tmp[16];
03012 
03013   /* extract the 16 bits from the OpenGL stipple pattern */
03014   for(n = 15; n >= 0; n--){
03015     tmp[n] = (char)(pattern & 0x01);
03016     pattern >>= 1;
03017   }
03018   /* compute the on/off pixel sequence */
03019   n = 0;
03020   for(i = 0; i < 8; i++){
03021     while(n < 16 && !tmp[n]){ off[i]++; n++; }
03022     while(n < 16 && tmp[n]){ on[i]++; n++; }
03023     if(n >= 15){ i++; break; }
03024   }
03025 
03026   /* store the on/off array from right to left, starting with off
03027      pixels. The PostScript specification allows for at most 11
03028      elements in the on/off array, so we limit ourselves to 5 on/off
03029      couples (our longest possible array is thus [on4 off4 on3 off3
03030      on2 off2 on1 off1 on0 off0]) */
03031   *nb = 0;
03032   for(n = i - 1; n >= 0; n--){
03033     array[(*nb)++] = factor * on[n];
03034     array[(*nb)++] = factor * off[n];
03035     if(*nb == 10) break;
03036   }
03037 }

03039 static int gl2psPrintPostScriptDash(GLushort pattern, GLint factor, char *str)
03040 {
03041   int len = 0, i, n, array[10];
03042 
03043   if(pattern == gl2ps->lastpattern && factor == gl2ps->lastfactor)
03044     return 0;
03045   
03046   gl2ps->lastpattern = pattern;
03047   gl2ps->lastfactor = factor;
03048   
03049   if(!pattern || !factor){
03050     /* solid line */
03051     len += gl2psPrintf("[] 0 %s\n", str);
03052   }
03053   else{
03054     gl2psParseStipplePattern(pattern, factor, &n, array);
03055     len += gl2psPrintf("[");
03056     for(i = 0; i < n; i++){
03057       if(i) len += gl2psPrintf(" ");
03058       len += gl2psPrintf("%d", array[i]);
03059     }
03060     len += gl2psPrintf("] 0 %s\n", str);
03061   }
03062   
03063   return len;
03064 }

03729 static void gl2psPDFgroupListWriteMainStream(void)
03730 {
03731   int i, j, lastel;
03732   GL2PSprimitive *prim = NULL, *prev = NULL;
03733   GL2PSpdfgroup *gro;
03734   GL2PStriangle t;
03735 
03736   if(!gl2ps->pdfgrouplist)
03737     return;
03738 
03739   for(i = 0; i < gl2psListNbr(gl2ps->pdfgrouplist); ++i){
03740     gro = (GL2PSpdfgroup*)gl2psListPointer(gl2ps->pdfgrouplist, i);
03741 
03742     lastel = gl2psListNbr(gro->ptrlist) - 1;
03743     if(lastel < 0)
03744       continue;
03745 
03746     prim = *(GL2PSprimitive**)gl2psListPointer(gro->ptrlist, 0);
03747 
03748     switch(prim->type){
03749     case GL2PS_POINT:
03750       gl2ps->streamlength += gl2psPrintf("1 J\n");
03751       gl2ps->streamlength += gl2psPrintPDFLineWidth(prim->width);
03752       gl2ps->streamlength += gl2psPrintPDFStrokeColor(prim->verts[0].rgba);
03753       for(j = 0; j <= lastel; ++j){  
03754         prim = *(GL2PSprimitive**)gl2psListPointer(gro->ptrlist, j);
03755         gl2ps->streamlength +=
03756           gl2psPrintf("%f %f m %f %f l\n",
03757                       prim->verts[0].xyz[0], prim->verts[0].xyz[1],
03758                       prim->verts[0].xyz[0], prim->verts[0].xyz[1]);
03759       }
03760       gl2ps->streamlength += gl2psPrintf("S\n"); 
03761       gl2ps->streamlength += gl2psPrintf("0 J\n");
03762       break;
03763     case GL2PS_LINE:
03764       /* We try to use as few paths as possible to draw lines, in
03765          order to get nice stippling even when the individual segments
03766          are smaller than the stipple */
03767       gl2ps->streamlength += gl2psPrintPDFLineWidth(prim->width);
03768       gl2ps->streamlength += gl2psPrintPDFStrokeColor(prim->verts[0].rgba);
03769       gl2ps->streamlength += gl2psPrintPostScriptDash(prim->pattern, prim->factor, "d");
03770       /* start new path */
03771       gl2ps->streamlength += 
03772         gl2psPrintf("%f %f m\n", 
03773                     prim->verts[0].xyz[0], prim->verts[0].xyz[1]);
03774       
03775       for(j = 1; j <= lastel; ++j){
03776         prev = prim;
03777         prim = *(GL2PSprimitive**)gl2psListPointer(gro->ptrlist, j);
03778         if(!gl2psSamePosition(prim->verts[0].xyz, prev->verts[1].xyz)){
03779           /* the starting point of the new segment does not match the
03780              end point of the previous line, so we end the current
03781              path and start a new one */
03782           gl2ps->streamlength += 
03783             gl2psPrintf("%f %f l\n", 
03784                         prev->verts[1].xyz[0], prev->verts[1].xyz[1]);
03785           gl2ps->streamlength += 
03786             gl2psPrintf("%f %f m\n", 
03787                         prim->verts[0].xyz[0], prim->verts[0].xyz[1]);
03788         }
03789         else{
03790           /* the two segements are connected, so we just append to the
03791              current path */
03792           gl2ps->streamlength += 
03793             gl2psPrintf("%f %f l\n",
03794                         prim->verts[0].xyz[0], prim->verts[0].xyz[1]);
03795         }
03796       }
03797       /* end last path */
03798       gl2ps->streamlength += 
03799         gl2psPrintf("%f %f l\n", 
03800                     prim->verts[1].xyz[0], prim->verts[1].xyz[1]);
03801       gl2ps->streamlength += gl2psPrintf("S\n");
03802       break;
03803     case GL2PS_TRIANGLE:
03804       gl2psFillTriangleFromPrimitive(&t, prim, GL_TRUE);
03805       gl2psSortOutTrianglePDFgroup(gro);
03806       
03807       /* No alpha and const color: Simple PDF draw orders  */
03808       if(t.prop & T_CONST_COLOR && t.prop & T_ALPHA_1){         
03809         gl2ps->streamlength += gl2psPrintPDFFillColor(t.vertex[0].rgba);        
03810         for(j = 0; j <= lastel; ++j){  
03811           prim = *(GL2PSprimitive**)gl2psListPointer(gro->ptrlist, j);
03812           gl2psFillTriangleFromPrimitive(&t, prim, GL_FALSE);
03813           gl2ps->streamlength 
03814             += gl2psPrintf("%f %f m\n"
03815                            "%f %f l\n"
03816                            "%f %f l\n"
03817                            "h f\n",
03818                            t.vertex[0].xyz[0], t.vertex[0].xyz[1],
03819                            t.vertex[1].xyz[0], t.vertex[1].xyz[1],
03820                            t.vertex[2].xyz[0], t.vertex[2].xyz[1]);
03821         }
03822       }
03823       /* Const alpha < 1 and const color: Simple PDF draw orders 
03824          and an extra extended Graphics State for the alpha const */
03825       else if(t.prop & T_CONST_COLOR && t.prop & T_ALPHA_LESS_1){               
03826         gl2ps->streamlength += gl2psPrintf("q\n"
03827                                            "/GS%d gs\n",
03828                                            gro->gsno);
03829         gl2ps->streamlength += gl2psPrintPDFFillColor(prim->verts[0].rgba);

04290 /* Put vertex' edge flag (8bit) and coordinates (32bit) in shader stream */
04291 
04292 static int gl2psPrintPDFShaderStreamDataCoord(GL2PSvertex *vertex, 
04293                                               size_t (*action)(unsigned long data, 
04294                                                                size_t size), 
04295                                               GLfloat dx, GLfloat dy, 
04296                                               GLfloat xmin, GLfloat ymin)
04297 {
04298   int offs = 0;
04299   unsigned long imap;
04300   GLfloat diff;
04301   double dmax = ~1UL;
04302   char edgeflag = 0;
04303 
04304   /* FIXME: temp bux fix for 64 bit archs: */
04305   if(sizeof(unsigned long) == 8) dmax = dmax - 2048.;
04306 
04307   offs += (*action)(edgeflag, 1);
04308 
04309   /* The Shader stream in PDF requires to be in a 'big-endian'
04310      order */
04311     
04312   if(GL2PS_ZERO(dx*dy)){
04313     offs += (*action)(0, 4);
04314     offs += (*action)(0, 4);
04315   }
04316   else{
04317     diff = (vertex->xyz[0] - xmin) / dx;
04318     if(diff > 1)
04319       diff = 1.0F;
04320     else if(diff < 0)
04321       diff = 0.0F;
04322     imap = (unsigned long)(diff * dmax);
04323     offs += (*action)(imap, 4);
04324       
04325     diff = (vertex->xyz[1] - ymin) / dy;
04326     if(diff > 1)
04327       diff = 1.0F;
04328     else if(diff < 0)
04329       diff = 0.0F;
04330     imap = (unsigned long)(diff * dmax);
04331     offs += (*action)(imap, 4);
04332   }
04333   
04334   return offs;
04335 }

05155 static void gl2psPrintSVGPrimitive(void *data)
05156 {
05157   GL2PSprimitive *prim;
05158   GL2PSxyz xyz[4];
05159   GL2PSrgba rgba[4];
05160   char col[32];
05161   int newline;
05162 
05163   prim = *(GL2PSprimitive**)data;
05164 
05165   if((gl2ps->options & GL2PS_OCCLUSION_CULL) && prim->culled) return;
05166 
05167   /* We try to draw connected lines as a single path to get nice line
05168      joins and correct stippling. So if the primitive to print is not
05169      a line we must first finish the current line (if any): */
05170   if(prim->type != GL2PS_LINE) gl2psEndSVGLine();
05171 
05172   gl2psSVGGetCoordsAndColors(prim->numverts, prim->verts, xyz, rgba);
05173 
05174   switch(prim->type){
05175   case GL2PS_POINT :
05176     gl2psSVGGetColorString(rgba[0], col);
05177     gl2psPrintf("<circle fill=\"%s\" ", col);
05178     if(rgba[0][3] < 1.0F) gl2psPrintf("fill-opacity=\"%g\" ", rgba[0][3]);
05179     gl2psPrintf("cx=\"%g\" cy=\"%g\" r=\"%g\"/>\n",
05180                 xyz[0][0], xyz[0][1], 0.5 * prim->width);
05181     break;
05182   case GL2PS_LINE :
05183     if(!gl2psSamePosition(gl2ps->lastvertex.xyz, prim->verts[0].xyz) ||
05184        !gl2psSameColor(gl2ps->lastrgba, prim->verts[0].rgba) ||
05185        gl2ps->lastlinewidth != prim->width ||
05186        gl2ps->lastpattern != prim->pattern ||
05187        gl2ps->lastfactor != prim->factor){
05188       /* End the current line if the new segment does not start where
05189          the last one ended, or if the color, the width or the
05190          stippling have changed (we will need to use multi-point
05191          gradients for smooth-shaded lines) */
05192       gl2psEndSVGLine();
05193       newline = 1;
05194     }
05195     else{
05196       newline = 0;
05197     }
05198     gl2ps->lastvertex = prim->verts[1];
05199     gl2psSetLastColor(prim->verts[0].rgba);
05200     gl2ps->lastlinewidth = prim->width;
05201     gl2ps->lastpattern = prim->pattern;
05202     gl2ps->lastfactor = prim->factor;
05203     if(newline){
05204       gl2psSVGGetColorString(rgba[0], col);
05205       gl2psPrintf("<polyline fill=\"none\" stroke=\"%s\" stroke-width=\"%g\" ", 
05206                   col, prim->width);
05207       if(rgba[0][3] < 1.0F) gl2psPrintf("stroke-opacity=\"%g\" ", rgba[0][3]);
05208       gl2psPrintSVGDash(prim->pattern, prim->factor);
05209       gl2psPrintf("points=\"%g,%g ", xyz[0][0], xyz[0][1]);
05210     }
05211     else{
05212       gl2psPrintf("%g,%g ", xyz[0][0], xyz[0][1]);
05213     }
05214     break;
05215   case GL2PS_TRIANGLE :
05216     gl2psPrintSVGSmoothTriangle(xyz, rgba);
05217     break;
05218   case GL2PS_QUADRANGLE :
05219     gl2psMsg(GL2PS_WARNING, "There should not be any quad left to print");
05220     break;
05221   case GL2PS_PIXMAP :
05222     gl2psPrintSVGPixmap(xyz[0][0], xyz[0][1], prim->data.image);
05223     break;
05224   case GL2PS_TEXT :
05225     gl2psSVGGetColorString(prim->verts[0].rgba, col);
05226     gl2psPrintf("<text fill=\"%s\" x=\"%g\" y=\"%g\" "
05227                 "font-size=\"%d\" font-family=\"%s\">%s</text>\n",
05228                 col, xyz[0][0], xyz[0][1],
05229                 prim->data.text->fontsize,
05230                 prim->data.text->fontname,
05231                 prim->data.text->str);
05232     break;
05233   case GL2PS_SPECIAL :
05234     /* alignment contains the format for which the special output text
05235        is intended */
05236     if(prim->data.text->alignment == GL2PS_SVG)
05237       gl2psPrintf("%s\n", prim->data.text->str);
05238     break;
05239   default :
05240     break;
05241   }
05242 }

05571

05575     if(prim->verts[i].xyz[0] > gl2ps->viewport[2])
05576       gl2ps->viewport[2] = (GLint)(prim->verts[i].xyz[0] + 0.5F);

05584 static GLint gl2psPrintPrimitives(void)
05585 {
05586   GL2PSbsptree *root;
05587   GL2PSxyz eye = {0.0F, 0.0F, 100.0F * GL2PS_ZSCALE};
05588   GLint used;
05589 
05590   used = glRenderMode(GL_RENDER);
05591 
05592   if(used < 0){
05593     gl2psMsg(GL2PS_INFO, "OpenGL feedback buffer overflow");
05594     return GL2PS_OVERFLOW;
05595   }
05596 
05597   if(used > 0)
05598     gl2psParseFeedbackBuffer(used);
05599 
05600   gl2psRescaleAndOffset();

05603     if(gl2psListNbr(gl2ps->primitives) && 
05604        (gl2ps->options & GL2PS_TIGHT_BOUNDING_BOX)){

05606       gl2ps->viewport[2] = gl2ps->viewport[3] = -100000;
05607       gl2psListAction(gl2ps->primitives, gl2psComputeTightBoundingBox);
05608     }

05614     /* empty feedback buffer and/or nothing else to print */

05623     gl2psListReset(gl2ps->primitives);
05624     break;

05628       gl2psListActionInverse(gl2ps->primitives, gl2psAddInImageTree);
05629       gl2psFreeBspImageTree(&gl2ps->imagetree);
05630     }
05631     gl2psListAction(gl2ps->primitives, gl2psbackends[gl2ps->format]->printPrimitive);
05632     gl2psListAction(gl2ps->primitives, gl2psFreePrimitive);
05633     /* reset the primitive list, waiting for the next viewport */
05634     gl2psListReset(gl2ps->primitives);
05635     break;

05637     root = (GL2PSbsptree*)gl2psMalloc(sizeof(GL2PSbsptree));
05638     gl2psBuildBspTree(root, gl2ps->primitives);

05641       gl2psTraverseBspTree(root, eye, -GL2PS_EPSILON, gl2psLess,
05642                            gl2psAddInImageTree, 1);
05643       gl2psFreeBspImageTree(&gl2ps->imagetree);
05644     }
05645     gl2psTraverseBspTree(root, eye, GL2PS_EPSILON, gl2psGreater, 
05646                          gl2psbackends[gl2ps->format]->printPrimitive, 0);
05647     gl2psFreeBspTree(&root);

05649        gl2psBuildBspTree) in case there is another viewport */
05650     gl2ps->primitives = gl2psListCreate(500, 500, sizeof(GL2PSprimitive*));
05651     break;
05652   }
05653   gl2psbackends[gl2ps->format]->printFinalPrimitive();
05654 
05655   return GL2PS_SUCCESS;
05656 }
